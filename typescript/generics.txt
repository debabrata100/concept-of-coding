# Define a Generic API Response Interface
interface ApiResponse<T> {
    success: boolean;
    message: string;
    data: T;
  }
type bodyType = {
  name: 'Deb'
}
# Create a Generic API Service Class
class ApiService {
  async get<T extends object>(url: string): Promise<ApiResponse<T>> {
    const response = await fetch(url);
    const json = await response.json();
    return json as ApiResponse<T>
  }
  
  async post<T extends object>(url: string, body: bodyType): Promise<ApiResponse<T>> {
 
    const response = await fetch(url, {
      method: 'POST',
      headers: {
        "Content-Type": "application/json"
      },
      body: JSON.stringify(body)
    });
    const json = await response.json();
    return json as ApiResponse<T>
  }
}

## Post some data
type NewUserRequest = {
    name: string;
    email: string;
  };
  
  async function createUser() {
    const newUser: NewUserRequest = {
      name: "John",
      email: "john@example.com"
    };
  
    const result = await api.post<User, NewUserRequest>(
      "https://api.example.com/users",
      newUser
    );
  
    if (result.success) {
      console.log("Created user ID:", result.data.id);
    } else {
      console.error("Error:", result.message);
    }
  }

# Make readonly to recursive object

type BuiltIn = Function | Date | Error | RegExp | Map<any, any> | Set<any>;

type DeepReadOnly<T> = 
T extends BuiltIn ? T : 
T extends Array<infer U> ? readonly DeepReadOnly<U>[] : 
T extends object ? {
  readonly [K in keyof T]: DeepReadOnly<T[K]>
}: T;
