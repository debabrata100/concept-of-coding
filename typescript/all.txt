// Pick<T, K>: Creates a new type by picking specific properties from an existing type.
const PublicUser: Pick<User, 'age' | 'name'> = {
    name: 'deb',
    age: 23
}

// Record<K, T>: reates a type with a set of keys of type K, where each key maps to a value of type T.

type Role = 'admin' | 'user' | 'guest';
const rolePermissions: Record<Role, string[]> = {
    admin: ['read', 'write', 'delete'],
    user: ['read'],
    guest: ['read']
}

// Omit<T, K>: Creates a new type by removing specific properties from an existing type.

type UserWithoutAge = Omit<User, 'age'>;

const u: UserWithoutAge = {
    name: 'deb',
    id: '234'
}

# Tuple
In TypeScript, a tuple is a typed array with a fixed number of elements, where each element can have a different type. Tuples are useful when you want to represent a collection of values where the order and type of each element are known and important.

function useUserData(): [User | null, Error | null, boolean] {
  const [user, setUser] = useState<User | null>(null);
  const [error, setError] = useState<Error | null>(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    fetch('/api/user')
      .then(res => res.json())
      .then(setUser)
      .catch(setError)
      .finally(() => setLoading(false));
  }, []);

  return [user, error, loading];
}

const [user, error, isLoading] = useUserData();
