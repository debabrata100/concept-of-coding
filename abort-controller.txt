The AbortController interface provides a way to signal that an asynchronous operation 
should be aborted. You can use it with the Fetch API to cancel a fetch request when 
needed. Here's an example implementation

// Create an AbortController instance
let abortController;

const fetchData = async (searchTerm, token) => {
  // If there is an existing request, abort it
  if (abortController) {
    abortController.abort();
  }

  // Create a new AbortController instance
  abortController = new AbortController();
  const signal = abortController.signal;

  try {
    const response = await fetch(`https://api.example.com/search?query=${searchTerm}`, { signal });
    const data = await response.json();

    // Check if the token matches the latest token
    if (token === latestToken) {
      console.log('Data received:', data);
      // Process the data
    } else {
      console.log('Discarding outdated results');
    }
  } catch (error) {
    if (error.name === 'AbortError') {
      console.log('Fetch aborted');
    } else {
      console.error('Error during fetch:', error);
    }
  }
};

// Debouncing function
const debounce = (func, delay) => {
  let timeoutId;

  return (...args) => {
    clearTimeout(timeoutId);
    timeoutId = setTimeout(() => {
      func(...args);
    }, delay);
  };
};

let latestToken = 0;

// Example usage
const debouncedFetchData = debounce((searchTerm) => {
  const token = ++latestToken;
  fetchData(searchTerm, token);
}, 500);

// Call debouncedFetchData when the search term changes
const searchTermInput = document.getElementById('searchTermInput');

searchTermInput.addEventListener('input', (event) => {
  const searchTerm = event.target.value;
  debouncedFetchData(searchTerm);
});
