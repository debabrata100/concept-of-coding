/*
If target is not found, it should return the correct position of target in the sorted array.
you should solve it in O(logn)
Example 1:
Input: nums = [1,3,5,6], target = 5
Output: 2
Example 2:
Input: nums = [1,3,5,6], target = 2
Output: 1
Example 3:
Input: nums = [1,3,5,6], target = 7
Output: 4
 
Constraints:
	•	1 <= nums.length <= 104
	•	-104 <= nums[i] <= 104
	•	nums contains distinct values sorted in ascending order.
	•	-104 <= target <= 104
*/

function findTargetPos(arr, target) {
    let n = arr.length;
    let low = 0;
    let high = n-1;
    while(low <= high) {
        let mid = ~~((low+high)/2);
        mid
        if(arr[mid] === target) return mid;
        if(target < arr[mid]) {
            high = mid - 1;
        }else{
            low = mid + 1;
        }
    }
    return low;
}
const nums = [1,3,5,6]; 
const target = 2;
console.log(findTargetPos(nums, target))
